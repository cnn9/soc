2. Infix to Postfix and Regex Match Example

import re

def infix_to_postfix(regex):
    if not regex:
        return ""
    
    # Handle parentheses
    if '(' in regex:
        left = regex.index('(')
        right = regex.index(')', left)
        inner = regex[left + 1:right]
        return regex[:left] + infix_to_postfix(inner) + regex[right + 1:]

    # Directly handle alphanumeric characters
    return ''.join(char for char in regex if char.isalnum())

# Get user input for regex and string
regex = input("Enter a regular expression: ")
string = input("Enter a string to match: ")

# Convert regex to postfix
postfix = infix_to_postfix(regex)

# Check if the string matches the regex
match_result = "ACCEPTED" if re.fullmatch(regex, string) else "REJECTED"

# Output results
print("Postfix form:", postfix)
print("Match:", match_result)

3. Quadruple Generation Example

def generate_quadruples(threeaddresscode):
    quadruples = []
    for line in threeaddresscode:
        if '=' in line:
            parts = line.split('=')
            result, expr = parts[0].strip(), parts[1].strip().split()
            if len(expr) == 3:
                quadruples.append((expr[1], expr[0], expr[2], result))  # Binary operation
            elif len(expr) == 2:
                quadruples.append((expr[0], expr[1], '', result))  # Unary operation
            elif len(expr) == 1:
                quadruples.append(('=', result, expr[0], ''))  # Assignment
        else:
            print(f"Error: Incorrect three-address code format in line: {line}")
    return quadruples

def print_quadruples(quadruples):
    print(f"{'Op':<5}{'Left':<5}{'Right':<5}{'Result':<7}")
    for op, Left, Right, result in quadruples:
        print(f"{op:<5}{Left:<5}{Right:<5}{result:<7}")

# Input
threeaddresscode_input = iter(lambda: input("Enter three-address code (end with 'end'): "), 'end')
quadruples = generate_quadruples(list(threeaddresscode_input))
print_quadruples(quadruples)

4. Time and Loop Example

import time
n = int(input("enter number"))

t1 = time.time()

for i in range(1, n + 1):
    print(i, end=" ") 

t2 = time.time()

t3 = time.time()

for i in range(1, n + 1):
    print(i, end=" ") 
    if i % 5 == 0:
        print()
