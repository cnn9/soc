MongoDB Commands

// Connect to your database

use myDatabase

// Insert sample documents into the 'products' collection

db.products.insertMany([
  { name: "Laptop", description: "Portable computer", price: 1200, tags: ["electronics", "computers", "portable"] },
  { name: "Smartphone", description: "Handheld device", price: 700, tags: ["electronics", "mobile", "handheld"] },
  { name: "Tablet", description: "Touchscreen device", price: 500, tags: ["electronics", "mobile", "touch"] }
]);

// Create a text index on the 'description' field

db.products.createIndex({ description: "text" });

// Create a compound index on 'name' and 'price'

db.products.createIndex({ name: 1, price: -1 });

// List all indexes on the 'products' collection

console.log("Indexes on 'products':", db.products.getIndexes());

// Perform a text search using the text index

var textQueryStats = db.products.find({ $text: { $search: "computer" } }).explain("executionStats");
console.log("Text Query Execution Stats:", textQueryStats);

// Use the compound index to find a product by name and sort by price

var compoundQueryStats = db.products.find({ name: "Laptop" }).sort({ price: -1 }).explain("executionStats");
console.log("Compound Query Execution Stats:", compoundQueryStats);

Redis Commands

// Connect to Redis

r = redis.Redis(host='localhost', port=6379, db=0)

// Store data in Redis

r.set('name', 'Alice')
r.setex('temp_data', 300, 'This is temporary data')

// Retrieve data from Redis

name = r.get('name')
if name:
    print("Retrieved name:", name.decode())

temp_data = r.get('temp_data')
if temp_data:
    print("Retrieved temporary data:", temp_data.decode())

// Redis CLI command to get the stored name

cmd get name
